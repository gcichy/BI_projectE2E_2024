#!/bin/bash

set -e

OVERWRITE=false

# Parse arguments
for arg in "$@"; do
    case $arg in
        --overwrite)
            OVERWRITE=true
            shift
            ;;
        *)
            echo "Unknown option: $arg"
            exit 1
            ;;
    esac
done

# Get directory of this script
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
TARGET_DIR="$HOME/temp"

# Create target directory if it doesn't exist
mkdir -p "$TARGET_DIR"

# Enable safe empty glob handling
shopt -s nullglob

if [ "$OVERWRITE" = true ]; then
    echo "Overwriting existing files..."
    cp -f "$SCRIPT_DIR"/* "$TARGET_DIR"/
else
    echo "Not overwriting existing files..."
    cp -n "$SCRIPT_DIR"/* "$TARGET_DIR"/
fi

echo "Done."
