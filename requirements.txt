<!-- Toast container (top-right) -->
<div class="toast-container position-fixed top-0 end-0 p-3" style="z-index: 1100;">
    <div id="statusToast" class="toast text-bg-primary" role="alert" aria-live="assertive" aria-atomic="true">
        <div class="d-flex">
            <div id="statusToastBody" class="toast-body">
                Status message here
            </div>
            <button type="button" class="btn-close btn-close-white me-2 m-auto"
                    data-bs-dismiss="toast" aria-label="Close"></button>
        </div>
    </div>
</div>

function showStatus(message, type="primary") {
    const toastEl = document.getElementById('statusToast');
    const toastBody = document.getElementById('statusToastBody');

    // update text & color
    toastBody.textContent = message;
    toastEl.className = `toast text-bg-${type}`;

    const toast = bootstrap.Toast.getOrCreateInstance(toastEl, {
        delay: 1000    // 1 second
    });
    toast.show();
}
