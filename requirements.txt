import xlwings as xw
import tempfile
from pathlib import Path

@xw.sub
def import_xlsx_bytes():
    wb = xw.Book.caller()
    dest_name = "ImportedData"

    # --- your bytes come from somewhere ---
    # xlsx_bytes = ...
    xlsx_bytes = get_xlsx_bytes_somehow()

    # 1) save to a temporary file
    tmp_path = None
    try:
        with tempfile.NamedTemporaryFile(delete=False, suffix=".xlsx") as f:
            f.write(xlsx_bytes)
            tmp_path = f.name

        # 2) open temp xlsx with Excel/xlwings
        src_wb = xw.Book(tmp_path)
        try:
            src_sht = src_wb.sheets[0]
            data = src_sht.used_range.value  # 2D list

            # 3) create/clear destination sheet in the calling xlsm
            if dest_name in [s.name for s in wb.sheets]:
                dest_sht = wb.sheets[dest_name]
                dest_sht.clear_contents()
            else:
                dest_sht = wb.sheets.add(dest_name, after=wb.sheets[-1])

            dest_sht.range("A1").value = data
            dest_sht.autofit()
        finally:
            src_wb.close()

    finally:
        # cleanup temp file
        if tmp_path:
            try:
                Path(tmp_path).unlink(missing_ok=True)
            except Exception:
                pass
